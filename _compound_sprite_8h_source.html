149  Component& addComponent(int priority, Component component) {
150  /* Moving the origin moves the drawn entity in the opposite direction.
151  * Move the origin of the component to the current position of the CompoundSprite
152  * and offset it by the origin of the CompoundSprite (this keeps things in the right place
153  * if the origin has been changed before the component is added).
154  *
155  * Setting the position of the component to the position of the compound sprite will then
156  * make the entity appear in the same place but rotate around the origin of the compound sprite
157  * instead of its own origin.
158  */
159  component.setOrigin(getPosition().x + getOrigin().x - component.getPosition().x, getPosition().y + getOrigin().y - component.getPosition().y);
160  component.setPosition(getPosition().x, getPosition().y);
161 
162 
163  // Add the component to the prioritizedComponents
164  auto it = m_prioritizedComponents.emplace(priority, std::make_unique<ComponentAdapter<Component>>(std::move(component)));
165  std::unique_ptr<InternalType>& returnValue = it->second;
166 
167  // Return the place in the components vector that the new component was placed.
168  return static_cast<Component&>(returnValue->getDataAsDrawable());
169  }

186  return &(underlyingData) == &componentToRemove;

188 
189  // If the drawable was found, erase it.
