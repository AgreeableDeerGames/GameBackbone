191  ProcessEventResult processEvent(sf::Int64 elapsedTime, const sf::Event& event)
192  {
193  // Exit early if not ready for input
194  if (!isReadyForInput())
195  {
196  // Return that the GestureMatchSignaler did not fire an action, is not ready for input, and the input was not consumed.
197  return { false, false, false };
198  }
199 
200  // Filter out events that this GestureMatchSignaler should ignore
201  if (!std::invoke(m_eventFilter, event))
202  {
203  // Return that the GestureMatchSignaler did not fire an action, its ready state, and the input was not consumed.
204  return { false, isReadyForInput(), false };
205  }
206 
207  // Process the input.
208  bool actionFired = false;
209  bool inputConsumed = false;
210  // If the events are the same and the time is within the maximum, increment the position and behave accordingly.
211  if (compareEvents(getNextEvent(), event) && elapsedTime < m_maxTimeBetweenInputs)
212  {
213  ++m_position;
214  // If the input was the last sf::Event in the Sequence, complete the match.
215  if (m_position == m_gesture.size())
216  {
217  completeMatch();
218  actionFired = true;
219  }
220  inputConsumed = true;
221  }
222  // The input did not match. Disable this GestureMatchSignaler.
223  else
224  {
225  m_position = 0;
226  m_readyForInput = false;
227  }
228 
229  // Return if the GestureMatchSignaler fired an action, its ready state, and if it consumed the input.
230  return { actionFired, isReadyForInput(), inputConsumed };
231  }
