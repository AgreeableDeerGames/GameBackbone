46 Point2D<int> ClusterGreenhouse::growCluster(Cluster* clusterToAddTo) {
47  // Create and iterator and put it to some random Point2D of the cluster's border
48  const std::set<Point2D<int>>* clusterBorderPointSet = clusterToAddTo->getBorderPointSet();
49  auto borderPointSetIter = clusterBorderPointSet->begin();
50  std::advance(borderPointSetIter, RandomGenerator.uniDist(0, (double)clusterBorderPointSet->size()));
51 
52  for (int i = 0; i < clusterBorderPointSet->size(); i++) {
53  // If the point is not already in a different cluster and also not on the edge
54  // of the graph, then it's okay to be added to this Cluster
55  if (pointToClusterMaps[pointToClusterMaps.size()-1].find(*borderPointSetIter) == pointToClusterMaps[pointToClusterMaps.size()-1].end() &&
56  graphDims.x > borderPointSetIter->x && graphDims.y > borderPointSetIter->y &&
57  0 <= borderPointSetIter->x && 0 <= borderPointSetIter->y) {
58  Point2D<int> pointToAdd = *borderPointSetIter;
59  clusterToAddTo->addPointToCluster(pointToAdd);
60 
61  pointToClusterMaps[pointToClusterMaps.size() - 1].insert(std::make_pair(pointToAdd, *clusterToAddTo));
62  return pointToAdd;
63  }
64  // Otherwise, check the next point in the border
65  else {
66  borderPointSetIter++;
67  if (borderPointSetIter == clusterBorderPointSet->end()) {
68  borderPointSetIter = clusterBorderPointSet->begin();
69  }
70  }
71  }
72  // If the there were no points which could be added to the cluster, we are given a flag Point2D{-1, -1}.
73  return Point2D<int>{-1, -1};

80 void ClusterGreenhouse::createClustersFromFrequencies(std::vector<double> frequencies) {
81  // If the input frequency vector is not empty...
82  if (!frequencies.empty()) {
83  std::multimap<Point2D<int>, Cluster > layerPointToClusterMap;
84  std::vector<Cluster> layerClusterVector;
85  double percentAdjustment = 0;
86  for (auto i = 0; i < frequencies.size(); i++) {
87  // make an origin for the cluster
88  Point2D<int> originPoint{ (int)RandomGenerator.uniDist(0, graphDims.x),(int)RandomGenerator.uniDist(0, graphDims.y) };
89 
90  // If that point is already in a different cluster, choose a different one
91  while (layerPointToClusterMap.find(originPoint) != layerPointToClusterMap.end()) {
92  originPoint = Point2D<int>{ (int)RandomGenerator.uniDist(0, graphDims.x),(int)RandomGenerator.uniDist(0, graphDims.y) };
93  }
94  // Adjust the frequency of the cluster
95  // When the percentages of each cluster are entered, they will be not relative to each other,
96  // ie .20, .10, .04
97  // They will be added to each other, so:
98  // .20, .30, .34
99  // sparsity is the leftover percent, so .16
100  percentAdjustment = frequencies[i] + percentAdjustment;
101  frequencies[i] = percentAdjustment;
102 
103  Cluster clusterToAdd(originPoint);
104  clusterToAdd.setClusterFrequency(frequencies[i]);
105  layerClusterVector.push_back(clusterToAdd);
106  layerPointToClusterMap.insert(std::make_pair(originPoint, clusterToAdd));
107  }
108  pointToClusterMaps.push_back(layerPointToClusterMap);
109  clusterVectors.push_back(layerClusterVector);
110  sparsity = 1 - frequencies[frequencies.size() - 1];
111  }
112  // if the frequencies vector is empty, generate a random one and then just call this function again.
113  else {
114  int numberOfClustersToMake = (int)RandomGenerator.uniDist(4, 8);
115 
116  // decide the frequency of this cluster based on how much of 0 through 1 (non-discrete) is left
117  double availablePercent = .15;
118  for (int i = 0; i < numberOfClustersToMake; i++) {
119  if (availablePercent < 0) {
120  break;
121  }
122 
123  double clusterFrequency = RandomGenerator.uniDist(0, availablePercent);
124  availablePercent -= clusterFrequency;
125 
126  // add the frequency
127  frequencies.push_back(clusterFrequency);
128  }
129  // call again with the newly generated percents
130  createClustersFromFrequencies(frequencies);
131  }
132 }
