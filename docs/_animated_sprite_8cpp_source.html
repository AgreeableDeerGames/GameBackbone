173 void AnimatedSprite::update(sf::Time currentTime) {
174 
175  if (animating && (currentTime.asMilliseconds() - lastUpdate.asMilliseconds() > (int)animationDelay)) {
176  switch (animationEnd) {
177  case ANIMATION_END_TYPE::ANIMATION_LOOP:
178  setCurrentFrame((currentFrame + 1) % currentAnimation->size());
179  break;
180  case ANIMATION_END_TYPE::ANIMATION_REVERSE:
181  // Only change the frame if the animation has more than one frame
182  if (currentAnimation->size() > 1) {
183  if (currentFrame >= currentAnimation->size() - 1 || (currentFrame == 0 && isReverse)) {
184  isReverse = !isReverse;
185  }
186 
187  if (!isReverse) {
188  setCurrentFrame(currentFrame + 1);
189  }
190  else {
191  setCurrentFrame(currentFrame - 1);
192  }
193  }
194  break;
195  case ANIMATION_END_TYPE::ANIMATION_STOP:
196  if (currentFrame < currentAnimation->size() - 1) {
197  setCurrentFrame(currentFrame + 1);
198  }
199  break;
200  }
201 
202  framesSpentInCurrentAnimation++;
203  setTextureRect(currentAnimation->at(currentFrame));
204  lastUpdate = currentTime;
205  }
