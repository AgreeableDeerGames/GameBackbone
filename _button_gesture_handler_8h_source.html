222  bool applyEventToOpenSet(sf::Int64 elapsedTime, const sf::Event& event)
223  {
224  bool eventConsumed = false;
225  for (std::size_t ii = 0; ii < m_openSetGestures.size(); ++ii)
226  {
227  // Forward event to the gesture bind
228  auto result = m_openSetGestures[ii]->processEvent(elapsedTime, event);
229 
230  // The bind cannot take any more input. Remove it from the open set.
231  if (!result.isReadyForInput)
232  {
233  m_openSetGestures.erase(m_openSetGestures.begin() + ii);
234  --ii;
235  }
236 
237  // The event was part of the gesture
238  if (result.inputConsumed)
239  {
240  eventConsumed = true;
