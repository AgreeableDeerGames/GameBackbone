261 void RelativeRotationSprite::initializeComponentVector(const std::vector<sf::Sprite*>& components, const std::vector<sf::Vector2f>& relativeOffsets) {
262  if (components.size() != relativeOffsets.size())
263  {
264  throw Error::RelativeRotationSprite_MismatchedSizes();
265  }
266 
267  // Add any components that are AnimatedSprites to AnimatedSprite storage to allow them to be updated
268  for (std::size_t ii = 0; ii < components.size(); ++ii) {
269  sf::Sprite* sprite = components[ii];
270  AnimatedSprite* animSprite = dynamic_cast<AnimatedSprite*>(sprite);
271  if (animSprite) {
272  addComponent(animSprite, relativeOffsets[ii]);
273  }
274  else {
275  addComponent(sprite, relativeOffsets[ii]);
276  }
277  }
278 }

286 void RelativeRotationSprite::initializeComponentVector(const std::vector<sf::Sprite*>& sprites, const std::vector<AnimatedSprite*>& animatedSprites, const std::vector<sf::Vector2f>& relativeOffsets) {
287  if (sprites.size() + animatedSprites.size() != relativeOffsets.size())

295  for (std::size_t ii = sprites.size(); ii < sprites.size() + animatedSprites.size(); ++ii) {
