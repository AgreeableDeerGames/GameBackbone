52 void Pathfinder::pathFind(const std::vector<PathRequest>& pathRequests, std::vector<std::deque<Point2D<int>>>* const returnedPaths) const {
53 
54  typedef std::pair<Point2D<int>, int> GridValuePair;
55 
56  //ensure that returned paths is big enough to store all results
57  returnedPaths->resize(pathRequests.size());
58 
59  //find result for each path request
60  for (unsigned int i = 0; i < pathRequests.size(); i++) {
61 
62  PathRequest pathRequest = pathRequests[i];
63 
64  //grid address of start and end points
65  Point2D<int> startPoint = pathRequest.start;
66  Point2D<int> endPoint = pathRequest.end;
67 
68  // the set of nodes already evaluated
69  std::set<Point2D<int>>* closedSet = new std::set<Point2D<int>>();
70 
71  //the set of currently discovered nodes that are already evaluated.
72  std::set<Point2D<int>>* openSet = new std::set<Point2D<int>>();
73  openSet->insert(startPoint);
74 
75  //cost to move to a point from the start
76  std::map<Point2D<int>, int>* score = new std::map<Point2D<int>, int>();
77  score->insert(GridValuePair(startPoint, 0));
78 
79  // For each node, which node it can most efficiently be reached from.
80  // If a node can be reached from many nodes, cameFrom will eventually contain the
81  // most efficient previous step.
82  std::map<Point2D<int>, Point2D<int>>* cameFrom = new std::map<Point2D<int>, Point2D<int>>();
83 
84  //search for path
85  (*returnedPaths)[i] = std::deque<Point2D<int>>(); //initialize path as empty
86  while (!openSet->empty()) {
87  //check current grid square
88  Point2D<int> current = chooseNextGridSquare(pathRequest, openSet, score);
89  if (current == endPoint) {
90  //reconstruct path, and add to output vector
91 
92  std::deque<Point2D<int>> inOrderPath = reconstructPath(endPoint, cameFrom);
93  (*returnedPaths)[i] = inOrderPath;
94  break;
95  }
96  openSet->erase(openSet->find(current));
97  closedSet->insert(current);
98 
99  //find neighbors
100  std::vector<Point2D<int>> neighbors = getNeighbors(current);
101 
102  for (Point2D<int> neighbor : neighbors) {
103  if (closedSet->find(neighbor) != closedSet->end()) {
104  continue;// no need to evaluate already evaluated nodes
105  }
106  //cost of reaching neighbor using current path
107  int transitionCost = (navigationGrid->at(current.x, current.y)->weight + navigationGrid->at(neighbor.x, neighbor.y)->weight) / 2;
108  int tentativeScore = score->at(current) + transitionCost;
109 
110  //discover new node
111  if (openSet->find(neighbor) == openSet->end()) {
112  //add blocked to closed set and unblocked to open set
113  if (navigationGrid->at(neighbor.x, neighbor.y)->weight >= BLOCKED_GRID_WEIGHT) {
114  closedSet->insert(neighbor);
115  continue;
116  }
117  else {
118  openSet->insert(neighbor);
119  }
120  }
121  else if (tentativeScore >= score->at(neighbor)) {
122  continue; // found a worse path
123  }
124 
125  //update or insert values for node
126  cameFrom->insert_or_assign(neighbor, current);
127  score->insert_or_assign(neighbor, tentativeScore);
128  }
129  }
130 
131  //free memory
132  delete closedSet;
133  delete openSet;
134  delete cameFrom;
135  delete score;
136  }
137 }

148 Point2D<int> Pathfinder::chooseNextGridSquare(const PathRequest & pathRequest, const std::set<Point2D<int>>* const availableGridSquares, std::map<Point2D<int>, int>* score) const {
149 
150  unsigned int shortestDistance = UINT_MAX;
151  Point2D<int> bestGridSquare = {-1, -1};
152 
153  for (const Point2D<int> gridSquare : *availableGridSquares) {
154  unsigned int gridSquareDistance = GB::calcSquaredDistance2D<int>(gridSquare, pathRequest.end);
155  gridSquareDistance *= score->at(gridSquare);
156  if (gridSquareDistance < shortestDistance) {
157  shortestDistance = gridSquareDistance;
158  bestGridSquare = gridSquare;
159  }
160  }

211  //add grid squares until the beginning (grid square that did not come from anywhere) is found
