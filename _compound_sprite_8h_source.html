171  Component& addComponent(int priority, Component component) {
172  /* Moving the origin moves the drawn entity in the opposite direction.
173  * Move the origin of the component to the current position of the CompoundSprite
174  * and offset it by the origin of the CompoundSprite (this keeps things in the right place
175  * if the origin has been changed before the component is added).
176  *
177  * Setting the position of the component to the position of the compound sprite will then
178  * make the entity appear in the same place but rotate around the origin of the compound sprite
179  * instead of its own origin.
180  */
181  component.setOrigin(getPosition().x + getOrigin().x - component.getPosition().x, getPosition().y + getOrigin().y - component.getPosition().y);
182  component.setPosition(getPosition().x, getPosition().y);
183 
184 
185  // Add the component to the prioritizedComponents
186  auto it = m_prioritizedComponents.emplace(priority, std::make_unique<ComponentAdapter<Component>>(std::move(component)));
187  std::unique_ptr<ComponentWrapper>& returnValue = it->second;
188 
189  // Return the place in the components vector that the new component was placed.
190  return static_cast<Component&>(returnValue->getDataAsDrawable());
191  }

203  // Find the drawable inside of the internal map.
204  auto it = std::find_if(m_prioritizedComponents.begin(), m_prioritizedComponents.end(),
205  [&componentToRemove](const std::pair<int, std::unique_ptr<ComponentWrapper>>& possibleRemoval) -> bool {
206  sf::Drawable& underlyingData = possibleRemoval.second->getDataAsDrawable();
207  return &(underlyingData) == &componentToRemove;
208  });
209 
210  // If the drawable was found, erase it.
211  if (it != m_prioritizedComponents.end())
212  {
213  m_prioritizedComponents.erase(it);
214  }
