51 void Pathfinder::pathFind(const std::vector<PathRequest>& pathRequests, std::vector<std::deque<sf::Vector2i>>* const returnedPaths) const {
52 
53  using GridValuePair = std::pair<sf::Vector2i, int>;
54 
55  //ensure that returned paths is big enough to store all results
56  returnedPaths->resize(pathRequests.size());
57 
58  //find result for each path request
59  for (unsigned int i = 0; i < pathRequests.size(); i++) {
60 
61  PathRequest pathRequest = pathRequests[i];
62 
63  //grid address of start and end points
64  sf::Vector2i startPoint = pathRequest.start;
65  sf::Vector2i endPoint = pathRequest.end;
66 
67  // the set of nodes already evaluated
68  std::set<sf::Vector2i, IsVector2Less<int>> closedSet;
69 
70  // the set of currently discovered nodes that are already evaluated.
71  std::set<sf::Vector2i, IsVector2Less<int>> openSet;
72  openSet.insert(startPoint);
73 
74  // cost to move to a point from the start
75  std::map<sf::Vector2i, int, IsVector2Less<int>> score;
76  score.insert(GridValuePair(startPoint, 0));
77 
78  // For each node, which node it can most efficiently be reached from.
79  // If a node can be reached from many nodes, cameFrom will eventually contain the
80  // most efficient previous step.
81  std::map<sf::Vector2i, sf::Vector2i, IsVector2Less<int>> cameFrom;
82 
83  //search for path
84  (*returnedPaths)[i] = std::deque<sf::Vector2i>(); //initialize path as empty
85  while (!openSet.empty()) {
86  //check current grid square
87  sf::Vector2i current = chooseNextGridSquare(pathRequest, openSet, score);
88  if (current == endPoint) {
89  //reconstruct path, and add to output vector
90 
91  std::deque<sf::Vector2i> inOrderPath = reconstructPath(endPoint, cameFrom);
92  (*returnedPaths)[i] = inOrderPath;
93  break;
94  }
95  openSet.erase(openSet.find(current));
96  closedSet.insert(current);
97 
98  //find neighbors
99  std::vector<sf::Vector2i> neighbors = getNeighbors(current);
100  for (sf::Vector2i neighbor : neighbors) {
101  if (closedSet.find(neighbor) != closedSet.end()) {
102  continue;// no need to evaluate already evaluated nodes
103  }
104  //cost of reaching neighbor using current path
105  int transitionCost = (navigationGrid->at(current.x, current.y)->weight + navigationGrid->at(neighbor.x, neighbor.y)->weight) / 2;
106  int tentativeScore = score.at(current) + transitionCost;
107 
108  //discover new node
109  if (openSet.find(neighbor) == openSet.end()) {
110  //add blocked to closed set and unblocked to open set
111  if (navigationGrid->at(neighbor.x, neighbor.y)->weight >= BLOCKED_GRID_WEIGHT) {
112  closedSet.insert(neighbor);
113  continue;
114  }
115  openSet.insert(neighbor);
116  }
117  else if (tentativeScore >= score.at(neighbor)) {
118  continue; // found a worse path
119  }
120 
121  //update or insert values for node
122  cameFrom.insert_or_assign(neighbor, current);
123  score.insert_or_assign(neighbor, tentativeScore);
124  }
125  }
126  }
127 }

178  }
