152 void CompoundSprite::removeComponent(sf::Sprite* component) {

154  // find the component and remove it from all pertinent vectors
155  auto it = std::find(components.begin(), components.end(), component);
156  if (it != components.end()) {
157  AnimatedSprite* animSprite = dynamic_cast<AnimatedSprite*>(*it);
