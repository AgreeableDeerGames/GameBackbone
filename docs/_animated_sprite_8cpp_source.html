164 void AnimatedSprite::update(sf::Time currentTime) {
165 
166  if (animating && (currentTime.asMilliseconds() - lastUpdate.asMilliseconds() > (int)animationDelay)) {
167  switch (animationEnd) {
168  case ANIMATION_END_TYPE::ANIMATION_LOOP:
169  setCurrentFrame((currentFrame + 1) % currentAnimation->size());
170  break;
171  case ANIMATION_END_TYPE::ANIMATION_REVERSE:
172  // Only change the frame if the animation has more than one frame
173  if (currentAnimation->size() > 1) {
174  if (currentFrame >= currentAnimation->size() - 1 || (currentFrame <= 0 && isReverse)) {
175  isReverse = !isReverse;
176  }
177 
178  if (!isReverse) {
179  setCurrentFrame(currentFrame + 1);
180  }
181  else {
182  setCurrentFrame(currentFrame - 1);
183  }
184  }
185  break;
186  case ANIMATION_END_TYPE::ANIMATION_STOP:
187  if (currentFrame < currentAnimation->size() - 1) {
188  setCurrentFrame(currentFrame + 1);
189  }
190  break;
191  }
192
