163 void AnimatedSprite::update(sf::Time currentTime) {
164 
165  if (animating && (currentTime.asMilliseconds() - lastUpdate.asMilliseconds() > (int)animationDelay)) {
166  switch (animationEnd) {
167  case ANIMATION_END_TYPE::ANIMATION_LOOP:
168  setCurrentFrame((currentFrame + 1) % currentAnimation->size());
169  break;
170  case ANIMATION_END_TYPE::ANIMATION_REVERSE:
171  // Only change the frame if the animation has more than one frame
172  if (currentAnimation->size() > 1) {
173  if (currentFrame >= currentAnimation->size() - 1 || (currentFrame == 0 && isReverse)) {
174  isReverse = !isReverse;
175  }
176 
177  if (!isReverse) {
178  setCurrentFrame(currentFrame + 1);
179  }
180  else {
181  setCurrentFrame(currentFrame - 1);
182  }
183  }
184  break;
185  case ANIMATION_END_TYPE::ANIMATION_STOP:
186  if (currentFrame < currentAnimation->size() - 1) {
187  setCurrentFrame(currentFrame + 1);
188  }
189  break;
190  }
